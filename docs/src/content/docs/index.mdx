## Features

<CardGrid>
  <Card title="Minimal API style" icon="rocket">
    Define jobs with delegates — same pattern as mapping endpoints. Parameters resolve from DI automatically.
  </Card>
  <Card title="Distributed" icon="random">
    Run multiple nodes. Every job executes exactly once — no double runs, no coordination code.
  </Card>
  <Card title="Real-time dashboard" icon="laptop">
    Built-in UI with live log streaming, progress tracking, job management, and node monitoring.
  </Card>
  <Card title="Streaming pipelines" icon="seti:pipeline">

    Return `IAsyncEnumerable<T>` from jobs and pipe data between stages as it's produced.
  </Card>
  <Card title="Batteries included" icon="puzzle">
    Retries, cron scheduling, timeouts, concurrency limits, lifecycle hooks, and OpenTelemetry — all built in.
  </Card>
</CardGrid>

## Example

export const sample = `using Surefire;
using Surefire.Dashboard;

var builder = WebApplication.CreateBuilder(args);

builder.Services.AddSurefire();

var app = builder.Build();

app.AddJob("Add", (int a, int b) => a + b);

app.AddJob("DataImport", async (JobContext ctx, ILogger<Program> logger, CancellationToken ct) =>
{
    for (var i = 1; i <= 10; i++)
    {
        logger.LogInformation("Step {Step}/10", i);
        await ctx.ReportProgressAsync(i / 10.0);
        await Task.Delay(1000, ct);
    }
}).WithDescription("Imports data and reports progress")
  .WithCron("* * * * *");

app.MapSurefireDashboard();

app.Run();`;
